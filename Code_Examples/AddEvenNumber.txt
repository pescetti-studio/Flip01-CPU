// A program that adds even numbers
X = 0
SUM = 0
index = 0

// initialize the sum to 0
load$ ax 0
store ax SUM

// initialize the index to 0
load$ ax 0
store ax index

// set the value X to 20
load$ ax 20
store ax X

SUM_LOOP:
load ax index
cmp ax X
jg END

// add the even number to the sum
load ax index
load bx SUM
addR
store ax SUM

// increment index by 2 to move to the next even number
load ax index
add$ ax 2
store ax index
jmp SUM_LOOP

END:
pause
